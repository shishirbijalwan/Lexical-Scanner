using namespace Scanner;

SemiExp::SemiExp(Toker* pToker) : _pToker(pToker) {}

bool SemiExp::get()
{
  if (_pToker == nullptr)
    throw(std::logic_error("no Toker reference"));
  _tokens.clear();
  while (true)
  {
    std::string token = _pToker->getTok();
    if (token == "")
      break;
    _tokens.push_back(token);
    
	if (token == "{") {
		return true;
	}
	if (token == ";") {
		return true;
	}
	if (token == "NEWLINE") {
		return true;
	}
	if (token == "public" || token == "protected") {
		token = _pToker->getTok();
		//std::cout << "Value of token inside " << token << "\n";
		_tokens.push_back(token);
		return true;
	}
	//std::cout << "Value of token " << token << "\n";
	if (token == "for") {
		do {
			token = _pToker->getTok();
			//std::cout << "Value of token inside " << token << "\n";
			_tokens.push_back(token);
		} while (token != "{");
		return true;
	}
	if (token == "#") {
		//std::cout << "Inside #";
		token = _pToker->getTok();
		_tokens.push_back(token);
		token = _pToker->getTok();

		//std::cout << "The value of the token is  " << token;
		if (token == "<") {
			_tokens.push_back(token);
			do {
				token = _pToker->getTok();
			//	std::cout << "Value of token inside " << token << "\n";
				_tokens.push_back(token);
			} while (token != ">");
		}else{
			_tokens.push_back("\"");
			//token = _pToker->getTok();
			//std::cout << "Value of token inside " << token << "\n";
		   _tokens.push_back(token);
			_tokens.push_back("\"");
		}
		return true;
	}
  }
  return false;
}

Token SemiExp::operator[](size_t n)
{
  if (n < 0 || n >= _tokens.size())
    throw(std::invalid_argument("index out of range"));
  return _tokens[n];
}

size_t SemiExp::length()
{
  return _tokens.size();
}

void SemiExp::show()
{
  std::cout << "\n  ";
  for (auto token : _tokens)
    if(token != "\n")
      std::cout << token << " ";
  std::cout << "\n";
}

int main()
{
  Toker toker;
  std::string fileSpec = "../../Updated_Project1/Tokenizer/check_file2.txt";
  std::fstream in(fileSpec);
  if (!in.good())
  {
    std::cout << "\n  can't open file " << fileSpec << "\n\n";
    return 1;
  }
  toker.attach(&in);

  SemiExp semi(&toker);
  while(semi.get())
  {
    std::cout << "\n  -- semiExpression --";
    semi.show();
  }
  /*
     May have collected tokens, but reached end of stream
     before finding SemiExp terminator.
   */
  if (semi.length() > 0)
  {
    std::cout << "\n  -- semiExpression --";
    semi.show();
    std::cout << "\n\n";
  }
  return 0;
}
